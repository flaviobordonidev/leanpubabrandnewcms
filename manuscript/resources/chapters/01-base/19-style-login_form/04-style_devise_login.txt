### Step 2 di sostituzione

{id="02-04-04_06", title=".../app/views/users/sessions/new.html.erb", lang=HTML+Mako, line-numbers=on, starting-line-number=8}
~~~~~~~~
          <%= form_for(resource, as: resource_name, url: session_path(resource_name), class: "form-signin") do |f| %>

            <div class="form-label-group">
              <%= f.email_field :email, autofocus: true, required: true, autocomplete: "email", id: "inputEmail", class: "form-control", placeholder: "Email address" %>
              <%= f.label :email, for: "inputEmail" %>
            </div>

            <div class="form-label-group">
              <%= f.password_field :password, autocomplete: "current-password", id: "inputPassword", class: "form-control", placeholder: "Password", required: true %>
              <%= f.label :password, for: "inputPassword" %>
            </div>

            <% if devise_mapping.rememberable? %>
              <div class="custom-control custom-checkbox mb-3">
                <%= f.check_box :remember_me, class: "custom-control-input", id: "customCheck1" %>
                <%= f.label :remember_me, for: "customCheck1", class: "custom-control-label" %>
              </div>
            <% end %>

            <div class="actions">
              <%= f.submit "Log in", class: "btn btn-lg btn-primary btn-block text-uppercase" %>
            </div>
